```tsx
"use client"

import { useScroll, useTransform, motion } from 'framer-motion'
import Image from 'next/image'
import { useEffect, useRef, useState } from 'react'

interface TextParallaxProps {
  text: string[]
  image: string[]
}

export default function TextParallax({ text, image }: TextParallaxProps) {
  const container = useRef<HTMLDivElement>(null)
  const [scrollProgress, setScrollProgress] = useState(0)
  const { scrollYProgress } = useScroll({
    target: container,
    offset: ['start end', 'end start']
  })

  useEffect(() => {
    const unsubscribe = scrollYProgress.onChange(setScrollProgress)
    return () => unsubscribe()
  }, [scrollYProgress])

  return (
    <div className="min-h-screen bg-gray-100 overflow-hidden">
      <div className="fixed top-0 left-0 p-4 bg-white z-50 text-black text-lg">
        Scroll Progress: {scrollProgress.toFixed(2)}
      </div>
      <div className="h-screen flex items-center justify-center text-black">
        <h1 className="text-5xl md:text-6xl lg:text-7xl font-bold text-center px-4">Scroll down to see the effect</h1>
      </div>
      <div ref={container} className="min-h-screen">
        <Slide text={text[0]} src={image[0]} direction="left" left="-40%" progress={scrollYProgress} />
        <Slide text={text[1]} src={image[1]} direction="right" left="-25%" progress={scrollYProgress} />
        <Slide text={text[2]} src={image[2]} direction="left" left="-75%" progress={scrollYProgress} />
      </div>
      <div className="h-screen flex items-center justify-center text-black">
        <h2 className="text-4xl md:text-5xl lg:text-6xl font-semibold text-center px-4">End of parallax effect</h2>
      </div>
    </div>
  )
}

interface SlideProps {
  src: string
  text: string
  direction: 'left' | 'right'
  left: string
  progress: any
}

const Slide = ({ src, text, direction, left, progress }: SlideProps) => {
  const translateX = useTransform(progress, [0, 1], [150 * (direction === 'left' ? -1 : 1), -150 * (direction === 'left' ? -1 : 1)])
  
  return (
    <motion.div
      style={{ x: translateX, left }}
      className="relative flex whitespace-nowrap my-40 md:my-48 lg:my-56"
    >
      <Phrase text={text} src={src} />
      <Phrase text={text} src={src} />
      <Phrase text={text} src={src} />
    </motion.div>
  )
}

interface PhraseProps {
  src: string
  text: string
}

const Phrase = ({ src, text }: PhraseProps) => {
  return (
    <div className="px-6 md:px-8 lg:px-10 flex gap-6 md:gap-8 lg:gap-10 items-center">
      <p className="text-4xl sm:text-5xl md:text-6xl lg:text-7xl font-serif text-black">{text}</p>
      <span className="relative h-[15vw] sm:h-[12vw] md:h-[10vw] lg:h-[8vw] aspect-[4/2] rounded-full overflow-hidden">
        <Image style={{ objectFit: "cover" }} src={src} alt="UI Design" fill sizes="(max-width: 768px) 100vw, (max-width: 1200px) 50vw, 33vw" />
      </span>
    </div>
  )
}
```
